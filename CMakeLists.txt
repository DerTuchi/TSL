cmake_minimum_required(VERSION 3.16)
project(tslgen)

include(tsl.cmake)

create_tsl(
	WORKAROUND_WARNINGS FALSE
	USE_CONCEPTS FALSE
	TSLGENERATOR_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
	DESTINATION ${GENERATOR_OUTPUT_PATH}
	TARGETS_FLAGS ${TSL_LSCPU_FLAGS}
	PRIMITIVES_FILTER ${TSL_PRIMITIVES_FILTER}
	DATATYPES_FILTER ${TSL_DATATYPES_FILTER}
	LINK_OPTIONS ${TSL_LINK_OPTIONS}
	GENERATOR_OPTIONS ${TSL_GENERATOR_OPTIONS}
)
message(STATUS "INCLUDE Directory: ${TSL_INCLUDE_DIRECTORY}")
